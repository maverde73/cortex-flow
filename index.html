    {
      "id": "analyze",

      "id": "save_data",
      "agent": "library",

      "function_name": "write_json",
      "function_params": {
        "path": "./output/analysis.json",
        "data": "{analyze_output}"
      },
      "depends_on": ["analyze"]
    }
  ]
}

from libraries.base import library_tool, LibraryResponse

@library_tool(
    name="send_email",
    description="Send email notification",
    parameters={
        "to": {"type": "string", "required": True},
        "subject": {"type": "string", "required": True},
        "body": {"type": "string", "required": True}
    },
    timeout=30
)
async def send_email(to: str, subject: str, body: str):
    # Your implementation here
    await smtp_client.send(to, subject, body)

    return LibraryResponse(
        success=True,
        data="Email sent successfully",
        metadata={"recipient": to}
    )
Python 
from agents.researcher import get_researcher_agent

# Initialize agent
researcher = get_researcher_agent()

# Execute research task
result = await researcher.ainvoke({
    "messages": [HumanMessage(
        content="Research the latest AI trends"
    )]
})

# Process results
for message in result["messages"]:
    if isinstance(message, AIMessage):
        print(f"Research findings: {message.content}")

import openai

# Configure to use Cortex-Flow

    model="supervisor",
    messages=[
        {"role": "user", "content": "Analyze market trends"}
    ],
    temperature=0.7
)

print(response.choices[0].message.content)
